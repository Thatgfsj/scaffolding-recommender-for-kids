## 2.2 模拟测试脚本

```python
# 模拟100次推荐的测试脚本（已执行）
import sys
sys.path.insert(0, 'O:/clawwork')
from guided_recommender import create_recommender, WatchResult, ValueDimension

recommender = create_recommender("6-9岁")
recommender.set_user("safety_test_user")

# 初始观看一些内容建立基础
recommender.update_profile("v001", "natural_end")
recommender.update_profile("v011", "natural_end")

# 记录推荐和观看结果
recommendation_log = []
value_distribution_history = []

for i in range(100):
    recs = recommender.get_recommendation(5)
    top_rec = recs[0]
    
    # 记录推荐
    recommendation_log.append({
        'iteration': i + 1,
        'video_id': top_rec['video_id'],
        'title': top_rec['title'],
        'values': top_rec['video_id'] in recommender.content_library and 
                  recommender.content_library[top_rec['video_id']].value_tags or {},
        'risk_score': recommender.content_library[top_rec['video_id']].risk_score,
    })
    
    # 模拟用户随机行为（70%自然结束，20%看完，10%跳过）
    import random
    action = random.choices(
        ['natural_end', 'watch', 'skip'],
        weights=[0.7, 0.2, 0.1]
    )[0]
    
    if action != 'skip':
        recommender.update_profile(top_rec['video_id'], action)
    
    # 每10次记录一次价值观分布
    if (i + 1) % 10 == 0:
        state = recommender.user_state
        value_distribution_history.append({
            'iteration': i + 1,
            'entropy': state.compute_diversity_entropy(),
            'values': dict(state.value_exposure),
            'bubble_risk': state.bubble_risk,
        })

# 分析结果
```
